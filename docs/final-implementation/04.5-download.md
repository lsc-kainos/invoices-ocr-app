# R5: Download

> Requisito: "Provide an option for users to download the uploaded documents with the appended extracted text and LLM interactions."

---

## Visão Geral

O download gera um arquivo ZIP contendo:

1. Documento original (como foi enviado)
2. Texto extraído (OCR)
3. Resumo narrativo
4. Transcrição do chat

---

## 1. Frontend

### 1.1 Botão de Download

```tsx
// apps/web/components/features/document-download/download-button.tsx
```

**Comportamento:**

- Status `READY` → botão ativo
- Outros status → botão desabilitado
- Clique → abre link de download em nova aba
- Loading → spinner durante geração

## 2. Backend

### 2.1 DownloadService

```typescript
// apps/api/src/download/download.service.ts
@Injectable()
export class DownloadService {
  async buildArchive(
    userId: string,
    documentId: string,
  ): Promise<{ stream: Readable; filename: string }> {
    // 1. Busca documento com ownership check
    const doc = await this.prisma.document.findFirst({
      where: { id: documentId, userId },
    });
    if (!doc) throw new NotFoundException();
    if (doc.status !== DocumentStatus.READY) {
      throw new ConflictException({ code: 'document_not_ready' });
    }

    // 2. Busca histórico de chat
    const session = await this.prisma.chatSession.findFirst({
      where: { userId, documentId: doc.id },
      include: {
        messages: {
          where: { role: { not: ChatRole.TOOL } },
          orderBy: { createdAt: 'asc' },
        },
      },
    });

    // 3. Lê arquivo original do storage
    const originalBuf = await this.storage.read(doc.storagePath);

    // 4. Gera arquivos auxiliares
    const extractedBuf = buildExtractedTextFile(doc.extractedText);
    const narrativeBuf = buildExtractedTextFile(narrative);
    const transcriptStr = buildChatTranscript(doc, session);

    // 5. Monta ZIP
    const archive = archiver('zip', { zlib: { level: 6 } });
    archive.append(originalBuf, { name: `original.${ext}` });
    archive.append(extractedBuf, { name: 'extracted-text.txt' });
    archive.append(narrativeBuf, { name: 'narrative.txt' });
    archive.append(transcriptStr, { name: 'chat-transcript.md' });
    await archive.finalize();

    return { stream: archive, filename: `${safeName}.zip` };
  }
}
```

### 2.2 Conteúdo do ZIP

| Arquivo                                          | Conteúdo                           |
| ------------------------------------------------ | ---------------------------------- |
| `original.pdf` / `original.jpg` / `original.png` | Documento como foi enviado         |
| `extracted-text.txt`                             | Texto bruto extraído pelo OCR      |
| `narrative.txt`                                  | Resumo narrativo gerado pelo LLM   |
| `chat-transcript.md`                             | Transcrição das interações no chat |

### 2.3 Chat Transcript

```typescript
// apps/api/src/download/helpers/chat-transcript.ts
export function buildChatTranscript(doc: Document, session: ChatSession | null): string {
  if (!session || session.messages.length === 0) {
    return `# Chat - ${doc.filename}\n\nNenhuma interação registrada.`;
  }

  return (
    `# Chat - ${doc.filename}\n\n` +
    session.messages
      .map((m) => {
        const role = m.role === 'USER' ? 'Usuário' : 'Assistente';
        return `## ${role} (${m.createdAt.toISOString()})\n\n${m.content}\n`;
      })
      .join('\n---\n\n')
  );
}
```

**Formato Markdown:** fácil de ler e processar.

### 2.4 Tratamento de Erros

```typescript
// Arquivo sumiu do storage
try {
  originalBuf = await this.storage.read(doc.storagePath);
} catch (err) {
  if (err.code === 'ENOENT') {
    // Marca documento como FAILED e retorna 404
    await this.prisma.document.update({
      where: { id: documentId },
      data: { status: DocumentStatus.FAILED, failureReason: 'storage_missing' },
    });
    throw new NotFoundException({ code: 'storage_missing' });
  }
  throw new ServiceUnavailableException({ code: 'storage_unavailable' });
}
```

**Design:** se o arquivo físico sumiu, o documento é marcado como FAILED para que o usuário saiba que algo deu errado.

## 3. Endpoints

### 3.1 API NestJS

```typescript
// apps/api/src/download/download.controller.ts
@Controller('api/v1/documents')
export class DownloadController {
  @Get(':id/download')
  @Throttle({ download: { ttl: 60_000, limit: 60 } })
  async download(@CurrentUser() user: User, @Param('id') id: string, @Res() res: Response) {
    const { stream, filename } = await this.download.buildArchive(user.id, id);
    res.setHeader('Content-Type', 'application/zip');
    res.setHeader('Content-Disposition', `attachment; filename="${filename}"`);
    stream.pipe(res);
  }
}
```

### 3.2 Rate Limiting

- Download: 60/min por usuário
- Throttle aplicado no endpoint
